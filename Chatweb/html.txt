  <script>
    const loginDiv = document.getElementById('login');
    const chatDiv = document.getElementById('chatbox');
    const usernameInput = document.getElementById('usernameInput');
    const startButton = document.getElementById('startChat');
    const chat = document.getElementById('chat');
    const form = document.getElementById('form');
    const input = document.getElementById('input');

    let users = {};
    let bannedUsers = new Set();
    let admins = new Set(); // wird dynamisch befÃ¼llt
    let username = '';

    const ADMIN_NAME = 'ErikTheOwner';
    const ADMIN_PASSWORD = 'SuperSecret123'; // <<< Ã„NDERE das Passwort hier, wenn du willst

    function appendMessage(msg, isAdmin = false) {
      const div = document.createElement('div');
      div.innerHTML = isAdmin
        ? `<span class="text-yellow-300 font-bold">${msg}</span>`
        : `<span class="text-white">${msg}</span>`;
      chat.appendChild(div);
      chat.scrollTop = chat.scrollHeight;
    }

    startButton.addEventListener('click', () => {
      const name = usernameInput.value.trim();
      if (!name) {
        alert("â— Bitte gib einen Namen ein!");
        return;
      }

      if (name === ADMIN_NAME) {
        const pw = prompt("ğŸ”’ Admin-Passwort eingeben:");
        if (pw !== ADMIN_PASSWORD) {
          alert("âŒ Falsches Passwort! Anmeldung abgebrochen.");
          return;
        }
        admins.add(ADMIN_NAME);
      }

      username = name;
      users[username] = { name: username };

      loginDiv.classList.add('hidden');
      chatDiv.classList.remove('hidden');
      appendMessage(`âœ… Willkommen, ${username}`, true);
    });

    form.addEventListener('submit', e => {
      e.preventDefault();
      const msg = input.value.trim();
      if (!msg) return;

      if (msg.startsWith('/ban ') && admins.has(username)) {
        const target = msg.split(' ')[1];
        bannedUsers.add(target);
        appendMessage(`ğŸš« ${target} wurde gebannt`, true);
        input.value = '';
        return;
      }

      if (msg.startsWith('/makeadmin ') && admins.has(username)) {
        const target = msg.split(' ')[1];
        admins.add(target);
        appendMessage(`ğŸ‘‘ ${target} ist jetzt Admin`, true);
        input.value = '';
        return;
      }

      if (bannedUsers.has(username)) {
        appendMessage(`âŒ Du bist gebannt.`, true);
      } else {
        const isAdmin = admins.has(username);
        appendMessage(`${isAdmin ? 'ğŸ‘‘ ' : ''}${username}: ${msg}`);
      }

      input.value = '';
    });
